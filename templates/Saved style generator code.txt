Sub Create_styles()
'
' Create_right_float_style Macro
'
'
' COMPLETE THE FOLLOWING FIELDS
style_type = "Fig"  ' Must be "Fig" or "Fig-cap", for figure or caption
operation = "create"    ' Must be "create" or "delete"
position_h = "center"  ' Must be "left", "center", or "right"
floats = False   ' True or False

frame_pad = 0.15 ' Distance from text
min_in = 1
max_in = 8
increment = 0.25

base_style = style_type
BasedUponStyle = style_type + "-template"
current_wd = min_in

If floats Then
    style_root = base_style + "-float-" + position_h + "-" ' Name of new style
Else
    style_root = base_style + "-" + position_h + "-"   'Name of new style
End If

Do While current_wd <= max_in
    wd_left = Trim(Str(Int(current_wd)))
    wd_rt = Trim(Str(current_wd - wd_left))
    If wd_rt = 0.25 Then
    wd_right = "25"
    ElseIf wd_rt = 0.5 Then
    wd_right = "5"
    ElseIf wd_rt = 0.75 Then
    wd_right = "75"
    Else: wd_right = "0"
    End If
    style_name = style_root + wd_left + "-" + wd_right
    
    On Error Resume Next
    ActiveDocument.Styles(style_name).Delete    ' Delete style
If operation = "create" Then
Set myStyle = ActiveDocument.Styles.Add(Name:=style_name, _
 Type:=wdStyleTypeParagraph)

 ActiveDocument.Styles.Add Name:=style_name, Type:=wdStyleTypeParagraph
    With ActiveDocument.Styles(style_name)
        .AutomaticallyUpdate = False
        .BaseStyle = BasedUponStyle
        .NextParagraphStyle = "Normal"
    End With
    With ActiveDocument.Styles(style_name).ParagraphFormat
        .LeftIndent = InchesToPoints(0)
        .RightIndent = InchesToPoints(0)
        .SpaceBefore = 0
        .SpaceBeforeAuto = False
        .SpaceAfter = 0
        .SpaceAfterAuto = False
        .LineSpacingRule = wdLineSpaceSingle
        .Alignment = wdAlignParagraphLeft
        .WidowControl = True
        If style_type = "Fig-cap" Then
            .KeepWithNext = True
            .KeepTogether = True
        Else
        .KeepWithNext = False
        .KeepTogether = False
        End If
        .PageBreakBefore = False
        .NoLineNumber = False
        .Hyphenation = True
        .FirstLineIndent = InchesToPoints(0)
        .OutlineLevel = wdOutlineLevelBodyText
        .CharacterUnitLeftIndent = 0
        .CharacterUnitRightIndent = 0
        .CharacterUnitFirstLineIndent = 0
        .LineUnitBefore = 0
        .LineUnitAfter = 0
        .MirrorIndents = False
        .TextboxTightWrap = wdTightNone
        .CollapsedByDefault = False
    End With
    ActiveDocument.Styles(style_name).NoSpaceBetweenParagraphsOfSameStyle = False
    ActiveDocument.Styles(style_name).ParagraphFormat.TabStops. _
        ClearAll
    ActiveDocument.Styles(style_name).LanguageID = wdEnglishUS
    ActiveDocument.Styles(style_name).NoProofing = False
    
    If floats Then   ' If floats, make a positioned frame
        With ActiveDocument.Styles(style_name).Frame
            .TextWrap = True
            .WidthRule = wdFrameExact
            .Width = InchesToPoints(current_wd)
            .HeightRule = wdFrameAuto
            If position_h = "left" Then
                .HorizontalPosition = wdFrameLeft
            ElseIf position_h = "right" Then
              .HorizontalPosition = wdFrameRight
            Else
              .HorizontalPosition = wdFrameCenter
            End If
            RelativeHorizontalPosition = wdRelativeHorizontalPositionColuffmn
            .VerticalPosition = InchesToPoints(frame_pad)
            .RelativeVerticalPosition = wdRelativeVerticalPositionParagraph
            .HorizontalDistanceFromText = InchesToPoints(frame_pad)
            .VerticalDistanceFromText = InchesToPoints(0)
            .LockAnchor = False
        End With
    End If
End If
current_wd = current_wd + increment
Loop

End Sub
